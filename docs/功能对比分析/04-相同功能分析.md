```typescript
// 运行长度编码 (RLE) - 算法完全相同
export class RLECompressor {
  static compress(data: Uint8Array): Uint8Array {
    const compressed: number[] = [];
    let currentByte = data[0];
    let count = 1;

    for (let i = 1; i < data.length; i++) {
      if (data[i] === currentByte && count < 255) {
        count++;
      } else {
        compressed.push(count, currentByte);
        currentByte = data[i];
        count = 1;
      }
    }
    compressed.push(count, currentByte);

    return new Uint8Array(compressed);
  }
}
```

```typescript
    sampleRate: number,
    uartData: ChannelData
  ): number {
    // 分析起始位持续时间
    const startBitDurations = this.findStartBits(uartData);

    // 计算最可能的波特率
    const avgDuration = this.calculateAverage(startBitDurations);
```

```typescript
// 循环缓冲区 - 实现原理相同
export class CircularBuffer<T> {
  private buffer: T[];
  private head: number = 0;
  private tail: number = 0;
  private size: number;

  constructor(size: number) {
    this.buffer = new Array(size);
    this.size = size;
  }

  push(item: T): void {
    this.buffer[this.tail] = item;
    this.tail = (this.tail + 1) % this.size;

    if (this.tail === this.head) {
      this.head = (this.head + 1) % this.size;
    }
  }
}
```
